```kotlin
@Composable
fun MiniBarChart(values: List<Double>, modifier: Modifier = Modifier.height(80.dp).fillMaxWidth()) {
    if (values.isEmpty()) return
    val max = values.maxOrNull() ?: 1.0
    Canvas(modifier = modifier) {
        val barWidth = size.width / values.size
        values.forEachIndexed { index, value ->
            val barHeight = (value / max) * size.height
            drawRect(
                topLeft = androidx.compose.ui.geometry.Offset(index * barWidth + 8f, size.height - barHeight),
                size = Size(barWidth - 16f, barHeight)
            )
        }
    }
```

# 10. Print-ready PDF for Yearly Giving Statements

## `StatementsPrinter.kt`

```kotlin
package com.gracechurch.gracefulgiving.util

import android.content.Context
import android.graphics.Paint
import android.graphics.pdf.PdfDocument
import android.os.Build
import java.io.File
import java.io.FileOutputStream
import java.text.SimpleDateFormat
import java.util.*

object StatementsPrinter {
    /**
     * Generate a simple multi-page PDF for yearly giving statements.
     * Returns the File where the PDF was written.
     */
    fun generateYearlyStatementsPdf(context: Context, statements: List<Pair<String, Double>>, year: Int): File {
        val pdf = PdfDocument()
        val pageInfo = PdfDocument.PageInfo.Builder(612, 792, 1).create() // Letter size

        val paint = Paint()
        val dateFormat = SimpleDateFormat("yyyy-MM-dd HH:mm", Locale.getDefault())
        val fileName = "yearly_statements_${year}_${dateFormat.format(Date())}.pdf"
        val outFile = File(context.cacheDir, fileName)

        // Simple pagination: 30 lines per page
        val linesPerPage = 30
        val pages = (statements.size + linesPerPage - 1) / linesPerPage

        for (pageIndex in 0 until pages) {
            val page = pdf.startPage(PdfDocument.PageInfo.Builder(612, 792, pageIndex + 1).create())
            val canvas = page.canvas
            var y = 40f
            paint.textSize = 14f
            canvas.drawText("Yearly Giving Statements - $year", 40f, y, paint)
            y += 20f
            canvas.drawText("Generated: ${dateFormat.format(Date())}", 40f, y, paint)
            y += 30f

            val start = pageIndex * linesPerPage
            val end = ((pageIndex + 1) * linesPerPage).coerceAtMost(statements.size)
            for (i in start until end) {
                val (name, total) = statements[i]
                canvas.drawText(String.format("%d. %s", i + 1, name), 40f, y, paint)
                canvas.drawText(String.format("$%,.2f", total), 380f, y, paint)
                y += 18f
            }

            pdf.finishPage(page)
        }

        FileOutputStream(outFile).use { fs ->
            pdf.writeTo(fs)
        }
        pdf.close()
        return outFile
    }
}
```

```xml
<provider
    android:name="androidx.core.content.FileProvider"
    android:authorities="${applicationId}.fileprovider"
    android:exported="false"
    android:grantUriPermissions="true">
    <meta-data
        android:name="android.support.FILE_PROVIDER_PATHS"
        android:resource="@xml/provider_paths"/>
</provider>
```

And `res/xml/provider_paths.xml`:

```xml
<?xml version="1.0" encoding="utf-8"?>
<paths xmlns:android="http://schemas.android.com/apk/res/android">
```
